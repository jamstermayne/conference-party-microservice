Usage Instructions

HTML (with data attributes)

<div data-icon="target" data-icon-size="48"></div>

JavaScript (programmatic)

import { getIcon } from './icon-gallery.js';

// Get icon as HTML string
const iconHtml = getIcon('handshake', 24, 'custom-class');
element.innerHTML = iconHtml;
